\section{Introduction}

COOL (Classroom Object-Oriented Language) contains a number of programming language features common in other object oriented languages (such as Java). 
\begin{itemize}
    \item \textbf{Pure OOP:} Strictly object-oriented and requires single inheritance and encapsulation.
    \item \textbf{Static Typing:} Uses type checking at compile time to make sure types are safe.
    \item \textbf{Expression-Based:} Most language constructs are expressions that return a value.
    \item \textbf{Minimalist Syntax:} On purpose, it leaves out complicated syntactic sugar in favor of simple logic.
    \item \textbf{Core Classes:} Uses a simple set of built-in classes that are the building blocks of the language (like Object, Int, and String).
\end{itemize}

\section{Context Free Grammar}
\subsection{Start Symbol}

\[
\text{Program} \rightarrow \text{ClassList}
\]

\subsection{Classes}

\begin{align*}
\text{ClassList} &\rightarrow \text{Class}\ \text{ClassList} \\
                 &\mid \text{Class} \\[6pt]
\text{Class} &\rightarrow \texttt{class}\ \text{TYPE}\ \{ \text{FeatureList} \}\ ; \\[6pt]
\text{FeatureList} &\rightarrow \text{Feature}\ \text{FeatureList} \\
                   &\mid \varepsilon \\[6pt]
\text{Feature} &\rightarrow \text{ID}\ ( \ ) : \text{TYPE}\ \{ \text{Expr} \}\ ;
\end{align*}

\subsection{Expressions}

\begin{align*}
\text{Expr} &\rightarrow \text{LetExpr} \\
            &\mid \text{AssignExpr} \\
            &\mid \text{WhileExpr} \\
            &\mid \text{IfExpr} \\
            &\mid \text{Block} \\
            &\mid \text{ArithExpr} \\
            &\mid \text{BoolExpr} \\
            &\mid \text{ID} \\
            &\mid \text{INT} \\
            &\mid \text{BOOL} \\
            &\mid \text{STRING} \\
            &\mid \text{FunctionCall}
\end{align*}

\subsection{Let Expression}

\begin{align*}
\text{LetExpr} &\rightarrow \texttt{let}\ \text{LetDeclList}\ \texttt{in}\ \text{Expr} \\[6pt]
\text{LetDeclList} &\rightarrow \text{LetDecl}\ ,\ \text{LetDeclList} \\
                   &\mid \text{LetDecl} \\[6pt]
\text{LetDecl} &\rightarrow \text{ID} : \text{TYPE} \\
               &\mid \text{ID} : \text{TYPE} \leftarrow \text{Expr}
\end{align*}

\subsection{Assignment}

\[
\text{AssignExpr} \rightarrow \text{ID} \leftarrow \text{Expr}
\]

\subsection{While Loop}

\[
\text{WhileExpr} \rightarrow \texttt{while}\ \text{Expr}\ \texttt{loop}\ \text{Expr}\ \texttt{pool}
\]

\subsection{If Statement}

\[
\text{IfExpr} \rightarrow 
\texttt{if}\ \text{Expr}\ \texttt{then}\ \text{Expr}\ \texttt{else}\ \text{Expr}\ \texttt{fi}
\]

\subsection{Block}

\begin{align*}
\text{Block} &\rightarrow \{ \text{ExprList} \} \\[6pt]
\text{ExprList} &\rightarrow \text{Expr}\ ; \\
                &\mid \text{Expr}\ ;\ \text{ExprList}
\end{align*}

\subsection{Arithmetic Expressions (With Precedence)}

\begin{align*}
\text{ArithExpr} &\rightarrow \text{ArithExpr} + \text{Term} \\
                 &\mid \text{ArithExpr} - \text{Term} \\
                 &\mid \text{Term} \\[6pt]
\text{Term} &\rightarrow \text{Term} * \text{Factor} \\
            &\mid \text{Term} / \text{Factor} \\
            &\mid \text{Factor} \\[6pt]
\text{Factor} &\rightarrow \text{ID} \\
              &\mid \text{INT} \\
              &\mid ( \text{Expr} )
\end{align*}

\subsection{Boolean Expressions}

\begin{align*}
\text{BoolExpr} &\rightarrow \text{Expr} < \text{Expr} \\
                &\mid \text{Expr}\ \texttt{and}\ \text{Expr} \\
                &\mid \texttt{not}\ \text{Expr} \\
                &\mid \text{BOOL}
\end{align*}

\subsection{Function Call}

\begin{align*}
\text{FunctionCall} &\rightarrow \text{ID} ( \text{ArgList} ) \\[6pt]
\text{ArgList} &\rightarrow \text{Expr}\ ,\ \text{ArgList} \\
               &\mid \text{Expr} \\
               &\mid \varepsilon
\end{align*}

\section{Setup Instructions}


\subsection*{Execution Steps}
\begin{enumerate}
    \item \textbf{Extract the Files:} Ensure the following files are located in your current working directory:
    \begin{itemize}
        \item \texttt{cool.l}
        \item \texttt{validprogram.cl}
        \item \texttt{invalidprogram.cl}
    \end{itemize}

    \item \textbf{Compile the Lex File:} Generate the C source code from the Lex specification by executing the following command:
\begin{verbatim}
lex cool.l
\end{verbatim}

    \item \textbf{Compile the C Code:} Compile the newly generated \texttt{lex.yy.c} file into an executable. You must link the Flex library using the \texttt{-ll} flag:
\begin{verbatim}
gcc lex.yy.c -o lexer -ll
\end{verbatim}

    \item \textbf{Run the Lexer:} The lexer is designed to process \texttt{validprogram.cl} by default. If you want to process a different file, provide it as a command-line argument. In both cases, it redirects the output to \texttt{output.txt}.
    
    To run with the default file (\texttt{validprogram.cl}):
\begin{verbatim}
./lexer
\end{verbatim}
    
    To run with a custom file (e.g., \texttt{invalidprogram.cl}):
\begin{verbatim}
./lexer invalidprogram.cl
\end{verbatim}

\end{enumerate}
